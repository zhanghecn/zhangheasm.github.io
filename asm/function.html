# 高级函数过程

# 堆栈传参

之前参数是寄存器上面,到由于用寄存器的太占位置，且容易更改到寄存器的值，所以后面 建议使用堆栈传参

一般来说ebp 是堆栈参数的基址 而esp则是堆栈的基址

  1. 参数 对于传参将指向rsp 栈基址 的上方 ,一下示例传参了 a和b 并且求和
addTwo proc
mov dword ptr [rsp+8],ecx;
mov dword ptr [rsp+10h],edx;  64位堆栈距离8字节 
push rbp; 
mov rbp,rsp;


movzx rax,word ptr [rbp+16]; push rbp后还得+8
movzx rbx,word ptr [rbp+24]; push rbp后还得+8
add rax,rbx;
pop rbp;
ret;
addTwo endp;

对于堆栈地址 越往上地址越大 切64位地址 需要+8个字节的地址符号 2. 局部变量 一般来说局部变量是存在栈空间的 也就是基于esp栈空间下的

localVariable proc
	push rbp;
	mov rbp,rsp;
	sub rsp,16; //堆栈扩展指针 16字节 准备填充局部变量 

	mov dword ptr [rbp - 8],3; 局部变量a
	mov dword ptr [rbp - 16],4; 局部变量b

	mov eax,dword ptr [rbp - 8]; 
	mov ebx,dword ptr [rbp - 16];

	mov rsp,rbp; 重置指针 删除局部变量 且pop rbp可还原

	add eax,ebx; a+b

	pop rbp; 

	ret;

localVariable endp

# LEA

LEA用来获取间接操作数地址 与OFFSET 类似 但OFFSET只能获取已知的.DATA 变量地址 而LEA 可以真正获取值的内存偏移地址

lea esi,[ebp + 16]
